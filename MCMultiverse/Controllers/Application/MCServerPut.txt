        // PUT: api/MCServers/5
        [HttpPut("{id}")]
        public async Task<IActionResult> PutMCServer([FromRoute] int id, [FromBody] MCServer mCServer)
        {
            if (!ModelState.IsValid)
            {
                return BadRequest(ModelState);
            }

            if (id != mCServer.Id)
            {
                return BadRequest();
            }

            var identity = (ClaimsIdentity)User.Identity;
            var claim = identity.FindFirst(ClaimTypes.NameIdentifier);
            var userId = claim.Value;

            if (_context.MCServers.FirstOrDefault(server => server.Id == id && server.Owner.Id == userId) == null 
                    && _context.MCServers.FirstOrDefault(server => server.Id == id) != null
            )
            {
                return NotFound();
            }

            _context.Entry(mCServer).State = EntityState.Modified;

            try
            {
                await _context.SaveChangesAsync();
            }
            catch (DbUpdateConcurrencyException)
            {
                if (!MCServerExists(id))
                {
                    return NotFound();
                }
                else
                {
                    throw;
                }
            }

            return NoContent();
        }
